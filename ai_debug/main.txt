def read_last_lines(path, n=200):
    if not os.path.exists(path):
        return f"File not found: {path}"
    try:
        with open(path, "r", errors="replace") as f:
            lines = f.readlines()
            return "".join(lines[-n:])
    except Exception as e:
        return f"Error reading {path}: {e}"

def get_screen_log():
    """Capture recent screen output via hardcopy"""
    try:
        subprocess.run(
            ["/usr/bin/screen", "-S", SCREEN_NAME, "-X", "hardcopy", "/tmp/alphabot_screen.txt"],
            timeout=5, capture_output=True
        )
        return read_last_lines("/tmp/alphabot_screen.txt", 150)
    except Exception as e:
        return f"Could not capture screen: {e}"

def get_db_summary():
    if not os.path.exists(DB_PATH):
        return {"error": f"DB not found at {DB_PATH}"}
    try:
        conn = sqlite3.connect(DB_PATH)
        conn.row_factory = sqlite3.Row
        cur = conn.cursor()
        result = {}

        # Recent trades
        try:
            cur.execute("SELECT * FROM trades ORDER BY created_at DESC LIMIT 10")
            result["recent_trades"] = [dict(r) for r in cur.fetchall()]
        except:
            result["recent_trades"] = []

        # Near misses
        try:
            cur.execute("SELECT * FROM near_misses ORDER BY created_at DESC LIMIT 10")
            result["near_misses"] = [dict(r) for r in cur.fetchall()]
        except:
            result["near_misses"] = []

        # Trade count today
        try:
            today = datetime.now().strftime("%Y-%m-%d")
            cur.execute("SELECT COUNT(*) as cnt FROM trades WHERE created_at LIKE ?", (f"{today}%",))
            result["trades_today"] = cur.fetchone()["cnt"]
        except:
            result["trades_today"] = "unknown"

        conn.close()
        return result
    except Exception as e:
        return {"error": str(e)}

def get_screen_status():
    try:
        r = subprocess.run(["/usr/bin/screen", "-ls"], capture_output=True, text=True, timeout=5)
        return r.stdout + r.stderr
    except Exception as e:
        return f"Error: {e}"

def build_debug_bundle():
    bundle = {
        "timestamp": datetime.now().isoformat(),
        "screen_status": get_screen_status(),
        "recent_log": get_screen_log(),
        "db": get_db_summary(),
    }
    return bundle

def call_claude(user_message: str, bundle: dict) -> str:
    if not CLAUDE_API_KEY:
        return "ERROR: CLAUDE_API_KEY not set in environment."
    try:
        client = anthropic.Anthropic(api_key=CLAUDE_API_KEY)
        bundle_text = json.dumps(bundle, indent=2, default=str)
        # Trim if too large
        if len(bundle_text) > 15000:
            bundle_text = bundle_text[:15000] + "\n... [truncated]"

        message = client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1000,
            system=ALPHABOT_CONTEXT,
            messages=[
                {
                    "role": "user",
                    "content": f"DEBUG BUNDLE:\n{bundle_text}\n\nQUESTION: {user_message}"
                }
            ]
        )
        return message.content[0].text
    except Exception as e:
        return f"Claude API error: {e}"
